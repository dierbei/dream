# 脚本
```shell
# 要删除的分支列表
$branches = @(
  "2600-add-rust-solutio",
  "2600-add-rust-solution",
  "2605-add-rust-solution",
  "2609-add-rust-solution",
  "2614-add-rust-solution",
  "2639-add-rust-solution",
  "2643-add-rust-solution",
  "2644-add-rust-solution"
)

# 删除本地分支
foreach ($branch in $branches) {
  git branch -D $branch
}

# 删除远程分支
foreach ($branch in $branches) {
  git push origin --delete $branch
}
```

# 运行脚本
```shell
.\delete-branches.ps1
```
