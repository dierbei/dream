# 脚本
```shell
# 要删除的分支列表
$branches = @(
  "2535-add-rust-solution"
  "2540-add-rust-solution"
  "2544-add-rust-solution"
  "2549-add-rust-solution"
  "2553-add-rust-solutio"
  "2553-add-rust-solution"
  "2558-add-rust-solutio"
  "2558-add-rust-solution"
)

# 删除本地分支
foreach ($branch in $branches) {
  git branch -D $branch
}

# 删除远程分支
foreach ($branch in $branches) {
  git push origin --delete $branch
}
```

# 运行脚本
```shell
.\delete-branches.ps1
```
