## k8s 集成

### wasm example
```yaml
apiVersion: node.k8s.io/v1
kind: RuntimeClass
metadata:
  name: kuasar-wasm-runtime
handler: wasm
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: wasm-example
spec:
  replicas: 1
  selector:
    matchLabels:
      app: wasm-example
  template:
    metadata:
      labels:
        app: wasm-example
    spec:
      runtimeClassName: kuasar-wasm-runtime
      containers:
        - name: wasm
          image: dierbei/csi_demo:rust-example-hello
          ports:
            - containerPort: 3030
---
apiVersion: v1
kind: Service
metadata:
  name: wasm-example-service
spec:
  selector:
    app: wasm-example
  ports:
    - name: port
      protocol: TCP
      port: 80
      targetPort: 3030
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: test-nginx
spec:
  replicas: 1
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
        - name: nginx
          image: nginx:latest
          ports:
            - containerPort: 80
```
```shell
kubectl exec -it <pod-name> -- curl wasm-example-service
```
