func merge(nums1 []int, m int, nums2 []int, n int)  {
    i, j, cur := m - 1, n - 1, n + m - 1

    for i >= 0 && j >= 0 {
        if nums1[i] > nums2[j] {
            nums1[cur] = nums1[i]
            i--
        } else {
            nums1[cur] = nums2[j]
            j--
        }
        cur--
    }

    for k := j; k >= 0; k-- {
        nums1[cur] = nums2[k]
        cur--
    }
}
