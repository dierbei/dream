package main

import "log"

var a [10]int
var tree [100][2]int
var idx int

func insert(v int) {
	p := 0

	for i := 30; i >= 0; i-- {
		if tree[p][v>>i&1] == 0 {
			idx++
			tree[p][v>>i&1] = idx
		}
		p = tree[p][v>>i&1]
	}
}

func query(v int) int {
	p := 0
	res := 0

	for i := 30; i >= 0; i-- {
		t := v >> i & 1
		if t == 0 {
			t = 1
		} else {
			t = 0
		}

		if tree[p][t] != 0 {
			res += 1 << i
			p = tree[p][t]
		} else {
			p = tree[p][v>>i&1]
		}
	}

	return res
}

func main() {
	//1 2 3
	insert(1)
	insert(2)
	insert(3)

	ans := 0
	for i := 1; i <= 3; i++ {
		ans = max(ans, query(i))
	}
	log.Println(ans)
}

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}
