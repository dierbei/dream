## 1. 代码
```go
package main

import "log"

var stk [10]int
var tt int

func main() {
	//3 4 2 7 5
	test(3)
	test(4)
	test(2)
	test(7)
	test(5)

}

func test(v int) {
	for tt > 0 && stk[tt] > v {
		tt--
	}
	if tt > 0 {
		log.Println(stk[tt])
	} else {
		log.Println(-1)
	}

	tt += 1
	stk[tt] = v
}
```
