func multiply(num1 string, num2 string) string {
    n, m := len(num1), len(num2)
    s1, s2 := make([]int, 0), make([]int, 0)

    for i := n - 1; i >= 0; i-- {
        s1 = append(s1, int(num1[i] - '0'))
    }
    for i := m - 1; i >= 0; i-- {
        s2 = append(s2, int(num2[i] - '0'))
    }

    s3 := make([]int, n + m)
    for i := 0; i < len(s1); i++ {
        for j := 0; j < len(s2); j++ {  
            s3[i + j] += s1[i] * s2[j]
        }
    }

    for i, t := 0, 0; i < len(s3); i++ {
        t += s3[i]
        s3[i] = t % 10
        t /= 10
    }

    k := len(s3) - 1
    for k > 0 && s3[k] == 0 {
        k--
    }
    
    ans := ""
    for i := k; i >= 0; i-- {
        ans += fmt.Sprintf("%d", s3[i])
    }

    return ans
}
