func searchRange(nums []int, target int) []int {
    if len(nums) == 0 {
        return []int{-1, -1}
    }

    l, r := 0, len(nums) - 1;
    for l < r {
        mid := (l + r) >> 1
        if nums[mid] >= target {
            r = mid
        } else {
            l = mid + 1
        }
    }

    if nums[r] != target {
        return []int{-1, -1}
    }

    L := r
    l, r = 0, len(nums) - 1
    for l < r {
        mid := (l + r + 1) >> 1
        if nums[mid] <= target {
            l = mid
        } else {
            r = mid - 1
        }
    }
    return []int{L, r}
}
