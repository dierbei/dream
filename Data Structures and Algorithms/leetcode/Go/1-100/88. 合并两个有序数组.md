func merge(nums1 []int, m int, nums2 []int, n int)  {
    k := m + n - 1
    m = m - 1
    n = n - 1

    for ; k >= 0 && m >= 0 && n >= 0; k-- {
        if nums1[m] <= nums2[n] {
            nums1[k] = nums2[n]
            n--
        } else {
            nums1[k] = nums1[m]
            m--
        }
    }

    for ; n >= 0; k, n = k - 1, n - 1 {
        nums1[k] = nums2[n]
    }


}
