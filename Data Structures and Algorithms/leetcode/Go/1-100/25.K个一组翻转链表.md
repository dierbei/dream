/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseKGroup(head *ListNode, k int) *ListNode {
    dummy := &ListNode{ Val: -1, Next: head }
    p := dummy
    
    for p != nil {
        q := p

        for i := 0; i < k && q != nil; i++ {
            q = q.Next
        }
        if q == nil {
            break
        }

        a, b := p.Next, p.Next.Next
        for i := 0; i < k - 1; i++ {
            c := b.Next
            b.Next = a
            a = b
            b = c
        }

        c := p.Next
        p.Next = a
        c.Next = b
        p = c
    }

    return dummy.Next
}
