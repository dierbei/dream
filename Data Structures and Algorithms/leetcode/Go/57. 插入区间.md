func insert(a [][]int, b []int) [][]int {
    ans, k := make([][]int, 0), 0

    for k <len(a) && a[k][1] < b[0] {
        ans = append(ans, a[k])
        k++
    }

    if k < len(a) {
        b[0] = min(b[0], a[k][0])

        for k < len(a) && a[k][0] <= b[1] {
            b[1] = max(b[1], a[k][1])
            k++
        }
    }
    ans = append(ans, b)

    for k < len(a) {
        ans = append(ans, a[k])
        k++
    }

    return ans
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}

func min(a, b int) int {
    if a < b {
        return a
    }
    return b
}
