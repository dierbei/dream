func semiOrderedPermutation(nums []int) int {
    i, j, n := 0, 0, len(nums)

    for k, v := range nums {
        if v == 1 {
            i = k
        }
        if v == n {
            j = k
        }
    }

    ans := i - 1 + n - j
    if i > j {
        ans = i - 1 + n - j - 1
    }

    return ans
}
