func rotate(nums []int, k int)  {
    k %= len(nums)

    var reverse func(nums []int, l, r int)
    reverse = func(nums []int, l, r int) {
        for i, j := l, r; i < j; i, j = i + 1, j - 1 {
            nums[i], nums[j - 1] = nums[j - 1], nums[i]
        }
    }

    reverse(nums, 0, len(nums))
    reverse(nums, 0, k)
    reverse(nums, k, len(nums))
}
