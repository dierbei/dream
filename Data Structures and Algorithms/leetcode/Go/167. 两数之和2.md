func twoSum(numbers []int, target int) []int {
    for i, j := 0, len(numbers) - 1; i < j; i++ {
        for i < j && numbers[i] + numbers[j] > target {
            j--
        }
        if numbers[i] + numbers[j] == target {
            return []int{i + 1, j + 1}
        }
    }
    return []int{}
}
