func getCommon(nums1 []int, nums2 []int) int {
    for _, x := range nums1 {
        for _, y := range nums2 {
            if x == y {
                return x
            }
            if y > x {
                break
            }
        }
    }

}

func getCommon(nums1 []int, nums2 []int) int {
    for i, j := 0, 0; i < len(nums1) && j < len(nums2); {
        if nums1[i] == nums2[j] {
            return nums1[i]
        }
        if nums1[i] < nums2[j] {
            i++
        } else nums1[i] > nums2[j] {
            j++
        }
    }

    return -1
}
