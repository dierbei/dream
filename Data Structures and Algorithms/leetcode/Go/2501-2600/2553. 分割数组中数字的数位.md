func separateDigits(nums []int) []int {
    ans := make([]int, 0)

    for i := 0; i < len(nums); i++ {
        tmp := make([]int, 0)
        for nums[i] != 0 {
            tmp = append(tmp, nums[i] % 10)
            nums[i] /= 10
        }

        for i := len(tmp) - 1; i >= 0; i-- {
            ans = append(ans, tmp[i])
        }
    }

    return ans
}
