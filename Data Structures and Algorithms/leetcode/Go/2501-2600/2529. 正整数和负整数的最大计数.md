func maximumCount(nums []int) int {
    pos, neg := 0, 0

    for _, n := range nums {
        if n > 0 {
            pos++
        } else if n < 0 {
            neg++
        }
    }

    return max(pos, neg)
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
