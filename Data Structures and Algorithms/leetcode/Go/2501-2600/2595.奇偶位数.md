func evenOddBit(n int) []int {
    even, odd := 0, 0

    for i := 0; n != 0; n, i = n >> 1, i ^ 1 {
        if (i == 0) {
            even += n & 1
        } else {
            odd += n & 1
        }
    }

    return []int{ even, odd }
}
