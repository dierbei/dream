func mergeArrays(nums1 [][]int, nums2 [][]int) [][]int {
    m := make(map[int]int)
    for _, arr := range nums1 {
        m[arr[0]] += arr[1]
    }
    for _, arr := range nums2 {
        m[arr[0]] += arr[1]
    }

    ans := make([][]int, 0)
    for k, v := range m {
        ans = append(ans, []int{ k, v })
    }

    sort.Slice(ans, func(i, j int) bool {
        return ans[i][0] < ans[j][0]
    })

    return ans
}

func mergeArrays(nums1 [][]int, nums2 [][]int) [][]int {
    cnt := make([]int, 1001)

    for _, arr := range nums1 {
        cnt[arr[0]] += arr[1];
    }

    for _, arr := range nums2 {
        cnt[arr[0]] += arr[1];
    }

    ans := make([][]int, 0)
    for i, v := range cnt {
        if v  > 0 {
            ans = append(ans, []int{ i, v })
        }
    }

}
