func findLHS(nums []int) int {
    m, ans := make(map[int]int), 0

    for _, n := range nums {
        m[n]++
    }

    for k, v := range m {
        if s, ok := m[k - 1]; ok {
            ans = max(ans, v + s)
        }
    }

    return ans
}

func max(a, b int) int {
    if a > b {
        return a 
    }
    return b
}
