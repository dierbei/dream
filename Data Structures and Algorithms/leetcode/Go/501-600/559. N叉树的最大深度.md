/**
 * Definition for a Node.
 * type Node struct {
 *     Val int
 *     Children []*Node
 * }
 */

func maxDepth(root *Node) int {
    ans := 0
    var dfs func(root *Node, depth int)
    dfs = func(root *Node, depth int) {
        if root == nil {
            return
        }
        ans = max(ans, depth)
        
        for _, c := range root.Children {
            dfs(c, depth + 1)
        }
    }

    dfs(root, 1)
    
    return ans
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
