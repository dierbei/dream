func checkRecord(s string) bool {
    a, l := 0, 0

    for _, c := range s {
        if c == 'A' {
            a++
            l = 0
        } else if c == 'L' {
            l++
        } else {
            l = 0
        }

        if a > 1 || l > 2 {
            return false
        }
    }

    return true
}
