func trap(height []int) int {
    stk := make([]int, 0)
    ans := 0

    for i := 0; i < len(height); i++ {
        last := 0

        for len(stk) > 0 && height[stk[len(stk) - 1]] <= height[i] {
            ans += (i - stk[len(stk) - 1] - 1) * (height[stk[len(stk) - 1]] - last);
            last = height[stk[len(stk) - 1]];
            stk = stk[:len(stk) - 1];
        }

        if len(stk) > 0 {
            ans += (i - stk[len(stk) - 1] - 1) * (height[i] - last)
        }

        stk = append(stk, i)
    }

    return ans
}
