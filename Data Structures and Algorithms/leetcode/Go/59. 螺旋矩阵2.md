func generateMatrix(n int) [][]int {
    ans := make([][]int, n)
    for i := 0; i < n; i++ {
        ans[i] = make([]int, n)
    }

    dx, dy := []int{0, 1, 0, -1}, []int{1, 0, -1, 0}
    for i, x, y, d := 1, 0, 0, 0; i <= n * n; i++ {
        ans[x][y] = i

        a, b := x + dx[d], y + dy[d]
        if a < 0 || a >= n || b < 0 || b >= n || ans[a][b] != 0 {
            d = (d + 1) % 4
            a, b = x + dx[d], y + dy[d]
        }

        x, y = a, b
    }

    return ans
}
