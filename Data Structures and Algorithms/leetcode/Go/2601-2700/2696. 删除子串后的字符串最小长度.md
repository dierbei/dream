func minLength(s string) int {
    b := []byte(s)
    ans := make([]byte, 0)

    for _, c := range b {
        if len(ans) > 0 && ans[len(ans) - 1] == 'A' && c == 'B' {
            ans = ans[:len(ans) - 1]
        } else if len(ans) > 0 && ans[len(ans) - 1] == 'C' && c == 'D' {
            ans = ans[:len(ans) - 1]
        } else {
            ans = append(ans, c)
        }
    }

    return len(ans)
}
