func findDuplicates(nums []int) []int {
    ans := make([]int, 0)


    for _, v := range nums {
        p := abs(v) - 1
        nums[p] *= -1

        if nums[p] > 0 {
            ans = append(ans, abs(v))
        }
    }

    return ans
}

func abs(a int) int {
    if a < 0 {
        return -a
    }
    return a
}
