func minMoves(nums []int) int {
    m := 1000000010  

    for _, v := range nums {
        m = min(m, v)
    }

    ans := 0
    for _, v := range nums {
        ans += v - m
    }

    return ans
}

func min(a, b int) int {
    if a < b {
        return a
    }
    return b
}
