func minSubArrayLen(target int, nums []int) int {
    ans := 100001

    for i, j, sum := 0, 0, 0; i < len(nums); i++ {
        sum += nums[i]

        for sum - nums[j] >= target {
            sum -= nums[j]
            j++
        }

        if sum >= target {
            ans = min(ans, i - j + 1)
        }
    }

    if ans == 100001 {
        ans = 0
    }

    return ans
}

func min(a, b int) int { 
    if a < b {
        return a
    }
    return b
}
