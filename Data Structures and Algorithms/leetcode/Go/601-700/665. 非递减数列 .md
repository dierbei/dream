func checkPossibility(nums []int) bool {
    n := len(nums)

    for i := 0; i < n - 1; i++ {
        if nums[i] > nums[i + 1] {
            a, b := nums[i], nums[i + 1]

            nums[i + 1] = a
            if check(nums) {
                return true
            }

            nums[i + 1] = b
            nums[i] = b
            if check(nums) {
                return true
            }

            return false
        }
    }

    return true
}

func check(nums []int) bool {
    n := len(nums)

    for i := 0; i < n - 1; i++ {
        if nums[i] > nums[i + 1] {
            return false
        }
    }

    return true
}
