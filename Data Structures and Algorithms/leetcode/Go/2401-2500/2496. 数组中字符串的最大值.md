```go
func maximumValue(strs []string) int {
    ans := 0

    for _, s := range strs {
        v, err := strconv.Atoi(s)
        if err != nil {
            ans = max(ans, len(s))
        } else {
            ans = max(ans, v)
        }
    }

    return ans
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
```

```go
func maximumValue(strs []string) int {
    parse := func(s string) int {
        x := 0

        for _, c := range s {
            if c >= 'a' && c <= 'z' {
                x = len(s)
                break
            }

            x = x * 10 + int(c - '0')
        }

        return x
    }

    ans := 0
    for _, s := range strs {
        if v := parse(s); v > ans {
            ans = v
        }
    }

    return ans
}
```
