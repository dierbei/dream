func combine(n int, k int) [][]int {
    ans, path := make([][]int, 0), make([]int, 0)

    var dfs func(n, k, start int)
    dfs = func(n, k, start int) {
        if k == 0 {
            tmp := make([]int, len(path))
            copy(tmp, path)
            ans = append(ans, tmp)
            return
        }

        for i := start; i <= n; i++ {
            path = append(path, i)
            dfs(n, k - 1, i + 1)
            path = path[:len(path) - 1]
        }
    }
    dfs(n, k, 1)

    return ans
}
