impl Solution {
    pub fn count_digits(num: i32) -> i32 {
        let mut ans = 0;
        let mut n = num;

        while n != 0 {
            if num % (n % 10) == 0 {
                ans += 1;
            }
            n /= 10
        }

        ans
    }   
}

impl Solution {
    pub fn count_digits(num: i32) -> i32 {
        num.to_string()
            .chars()
            .filter(|&c| c != '0')
            .filter(|&c| num % c.to_digit(10).unwrap() as i32 == 0)
            .count() as i32
    }   
}
