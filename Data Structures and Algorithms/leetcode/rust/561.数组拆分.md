impl Solution {
    pub fn array_pair_sum(nums: Vec<i32>) -> i32 {
        let mut nums = nums;
        nums.sort();

        let mut ans = 0;

        let mut i = 0;
        loop {
            if i >= nums.len() {
                break;
            }

            ans += nums[i];

            i += 2;
        }

        ans
    }
}
