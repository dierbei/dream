class Solution {
public:
    vector<vector<int>> insert(vector<vector<int>>& intervals, vector<int>& newInterval) {
        vector<vector<int>> ans;

        int k = 0;
        while (k < intervals.size() && intervals[k][1] < newInterval[0]) ans.push_back(intervals[k++]);

        if (k < intervals.size()) {
            newInterval[0] = min(newInterval[0], intervals[k][0]);
            while (k < intervals.size() && intervals[k][0] <= newInterval[1]) newInterval[1] = max(newInterval[1], intervals[k++][1]);
        }

        ans.push_back(newInterval);

        while (k < intervals.size()) ans.push_back(intervals[k++]);

        return ans;
    }
};
