class Solution {
public:
    bool checkValidString(string s) {
        int l = 0, r = 0;

        for (auto c : s) {
            if (c == '(') l++, r++;
            else if (c == ')') l--, r--;
            else l--, r++;

            l = max(l, 0);

            if (l > r) return false;
        }

        return l == 0;
    }
};
